function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var resultado = {}; var hCartera = ss.getSheetByName('CARTERA'); var posiciones = []; if (hCartera) { var lastRow = hCartera.getLastRow(); for (var r = 3; r < lastRow; r++) { var nombre = hCartera.getRange(r, 1).getValue(); if (!nombre || nombre === 'TOTALES') break; var nombreStr = String(nombre); var pActualRaw = Number(hCartera.getRange(r, 5).getValue()) || 0; if (nombreStr.indexOf('MSCI World') >= 0 && pActualRaw > 500) { pActualRaw = pActualRaw / 100; } posiciones.push({ nombre: nombreStr, acciones: Number(hCartera.getRange(r, 3).getValue()) || 0, pCompra: Number(hCartera.getRange(r, 4).getValue()) || 0, pActual: pActualRaw, valor: Number(hCartera.getRange(r, 6).getValue()) || 0, gp: Number(hCartera.getRange(r, 7).getValue()) || 0, rent: Number(hCartera.getRange(r, 8).getValue()) || 0, divAcc: Number(hCartera.getRange(r, 9).getValue()) || 0, divAnio: Number(hCartera.getRange(r, 11).getValue()) || 0, divMes: Number(hCartera.getRange(r, 12).getValue()) || 0 }); } var totRow = lastRow; resultado.totales = { valorCartera: Number(hCartera.getRange(totRow, 6).getValue()) || 0, gpTotal: Number(hCartera.getRange(totRow, 7).getValue()) || 0, divAnioTotal: Number(hCartera.getRange(totRow, 11).getValue()) || 0, divMesTotal: Number(hCartera.getRange(totRow, 12).getValue()) || 0 }; } resultado.posiciones = posiciones; var hTrampas = ss.getSheetByName('TRAMPAS'); var trampas = []; if (hTrampas) { var lastT = hTrampas.getLastRow(); for (var r = 3; r <= lastT; r++) { var nombreT = hTrampas.getRange(r, 1).getValue(); if (!nombreT) continue; trampas.push({ nombre: String(nombreT), calidad: String(hTrampas.getRange(r, 2).getValue()), prioridad: String(hTrampas.getRange(r, 3).getValue()), trampa: Number(hTrampas.getRange(r, 4).getValue()) || 0, pActual: Number(hTrampas.getRange(r, 5).getValue()) || 0, distancia: Number(hTrampas.getRange(r, 6).getValue()) || 0, acciones: Number(hTrampas.getRange(r, 7).getValue()) || 0, capital: Number(hTrampas.getRange(r, 8).getValue()) || 0, divAcc: Number(hTrampas.getRange(r, 9).getValue()) || 0, pago: String(hTrampas.getRange(r, 11).getValue()) }); } } resultado.trampas = trampas; // ═══ OPERACIONES CERRADAS (FIFO) + COMISIONES EN POSICIONES ABIERTAS, desde TR_DATOS ═══ var hOps = ss.getSheetByName('TR_DATOS'); var operaciones = []; var costeVendidoTotal = 0; var comisionesAbiertas = {}; function limpiarTexto(v) { return String(v || '').replace(/[^A-Za-z0-9]/g, '').toUpperCase(); } if (hOps && hOps.getLastRow() > 1) { var lastO = hOps.getLastRow(); var datos = hOps.getRange(2, 1, lastO - 1, 12).getValues(); // columnas A:L var filas = []; for (var i = 0; i < datos.length; i++) { var fila = datos[i]; var category = limpiarTexto(fila[3]); // D=category var type = limpiarTexto(fila[4]); // E=type if (category !== 'TRADING' || (type !== 'BUY' && type !== 'SELL')) continue; var sharesRaw = Math.abs(parseFloat(String(fila[8]).replace(',', '.')) || 0); // I=shares var priceRaw = Math.abs(parseFloat(String(fila[9]).replace(',', '.')) || 0); // J=price if (!sharesRaw || !priceRaw) continue; var fechaVal = fila[1]; // B=date var fechaStr = (Object.prototype.toString.call(fechaVal) === '[object Date]') ? Utilities.formatDate(fechaVal, Session.getScriptTimeZone(), 'yyyy-MM-dd') : String(fechaVal); filas.push({ datetime: fila[0], // A=datetime (para ordenar con precisión) date: fechaStr, name: String(fila[6] || ''), // G=name symbol: String(fila[7] || ''), // H=symbol shares: sharesRaw, type: type, price: priceRaw, fee: Math.abs(parseFloat(String(fila[11]).replace(',', '.')) || 0) // L=fee }); } filas.sort(function(a, b) { return new Date(a.datetime) - new Date(b.datetime); }); var lotes = {}; filas.forEach(function(r) { var sym = r.symbol || r.name; if (!sym) return; if (!lotes[sym]) lotes[sym] = []; if (r.type === 'BUY') { lotes[sym].push({ shares: r.shares, price: r.price, feePorAccion: r.fee / r.shares, fecha: r.date, nombre: r.name }); } else if (r.type === 'SELL') { var restante = r.shares, coste = 0, accEmparejadas = 0; while (restante > 1e-8 && lotes[sym] && lotes[sym].length) { var lote = lotes[sym][0]; var usar = Math.min(lote.shares, restante); coste += usar * (lote.price + lote.feePorAccion); accEmparejadas += usar; lote.shares -= usar; restante -= usar; if (lote.shares <= 1e-8) lotes[sym].shift(); } var ingreso = r.shares * r.price - r.fee; var gp = ingreso - coste; costeVendidoTotal += coste; operaciones.push({ n: r.name, fecha: r.date, acciones: r.shares, pVenta: r.price, pCompraMedio: accEmparejadas ? (coste / accEmparejadas) : 0, coste: coste, ingreso: ingreso, gp: gp, gpPct: coste ? (gp / coste * 100) : 0 }); } }); operaciones.sort(function(a, b) { return new Date(b.fecha) - new Date(a.fecha); }); // Lo que queda sin consumir en "lotes" son tus posiciones ABIERTAS ahora mismo. // Sumamos, por activo, cuánto de comisión de compra sigue "viva" en esas acciones. Object.keys(lotes).forEach(function(sym) { var accionesRestantes = 0, comisionTotal = 0, nombre = sym; lotes[sym].forEach(function(lote) { accionesRestantes += lote.shares; comisionTotal += lote.shares * lote.feePorAccion; if (lote.nombre) nombre = lote.nombre; }); if (accionesRestantes > 1e-8) { comisionesAbiertas[nombre] = { acciones: accionesRestantes, comision: comisionTotal }; } }); } resultado.operaciones = operaciones; resultado.costeVendidoTotal = costeVendidoTotal; resultado.comisionesAbiertas = comisionesAbiertas; // ═══ FIN OPERACIONES CERRADAS ═══ resultado.timestamp = new Date().toISOString(); resultado.ok = true; return ContentService .createTextOutput(JSON.stringify(resultado)) .setMimeType(ContentService.MimeType.JSON); }
